  <div class="page">

    <header class="header">

      <h1 class="title">TrueScope</h1>


      <div class="tabs" role="tablist" aria-label="Input type">

        <button class="tab active" id="tabText" type="button" aria-selected="true">Text</button>

        <button class="tab" id="tabImage" type="button" aria-selected="false">Image</button>

        <button class="tab" id="tabVideo" type="button" aria-selected="false">Video</button>

      </div>

    </header>


    <main class="panel">

      <!-- Text Mode -->

      <section id="textSection" class="mode">

        <input type="text" id="claimInput" placeholder="Enter a claim to verify..." />

        <button id="checkBtn" class="primary">Check Claim</button>

        <p class="subtle">Tip: Keep claims short for best results.</p>

      </section>


      <!-- Image Mode (UI only for now) -->

      <section id="imageSection" class="mode hidden">

        <div class="uploadBox">

          <p class="uploadTitle">Upload an image to extract a claim (coming next)</p>

          <input type="file" id="imageFile" accept="image/*" />

          <button class="primary" id="imageBtn" disabled>Check Image</button>

          <p class="subtle">This endpoint is not implemented yet.</p>

        </div>

      </section>


      <!-- Video Mode (UI only for now) -->

      <section id="videoSection" class="mode hidden">

        <div class="uploadBox">

          <p class="uploadTitle">Upload a video to extract a claim (coming next)</p>

          <input type="file" id="videoFile" accept="video/*" />

          <button class="primary" id="videoBtn" disabled>Check Video</button>

          <p class="subtle">This endpoint is not implemented yet.</p>

        </div>

      </section>


      <div id="results"></div>

  <script>

    const API_BASE = "http://127.0.0.1:8000";


    const tabText = document.getElementById("tabText");

    const tabImage = document.getElementById("tabImage");

    const tabVideo = document.getElementById("tabVideo");


    const textSection = document.getElementById("textSection");

    const imageSection = document.getElementById("imageSection");

    const videoSection = document.getElementById("videoSection");


    function setActiveTab(active) {

      const all = [tabText, tabImage, tabVideo];

      all.forEach(btn => btn.classList.remove("active"));

      tabText.setAttribute("aria-selected", active === "text");

      tabImage.setAttribute("aria-selected", active === "image");

      tabVideo.setAttribute("aria-selected", active === "video");


      textSection.classList.toggle("hidden", active !== "text");

      imageSection.classList.toggle("hidden", active !== "image");

      videoSection.classList.toggle("hidden", active !== "video");


      if (active === "text") tabText.classList.add("active");

      if (active === "image") tabImage.classList.add("active");

      if (active === "video") tabVideo.classList.add("active");

    }


    tabText.addEventListener("click", () => setActiveTab("text"));

    tabImage.addEventListener("click", () => setActiveTab("image"));

    tabVideo.addEventListener("click", () => setActiveTab("video"));


    // Rating pill style helper

    function ratingClass(ratingText) {

      const r = (ratingText || "").toLowerCase();


      if (r.includes("true") && !r.includes("unverified") && !r.includes("mostly false")) return "rating-true";

      if (r.includes("false") || r.includes("incorrect") || r.includes("inaccurate")) return "rating-false";

      if (r.includes("mixed") || r.includes("misleading") || r.includes("mostly")) return "rating-mixed";

      return "rating-unknown";

    }


    function safeShortRating(r) {

      const txt = (r || "Unverified").toString();

      return txt.split(".")[0].slice(0, 40);

    }


    async function checkClaim() {

      const claim = document.getElementById("claimInput").value.trim();

      const resultsDiv = document.getElementById("results");


      if (!claim) {

        resultsDiv.innerHTML = `<p class="no-result">Please enter a claim to verify.</p>`;

        return;

      }


      resultsDiv.innerHTML = `<p class="loading">Checking...</p>`;


      try {

        const response = await fetch(`${API_BASE}/fact-check`, {

          method: "POST",

          headers: { "Content-Type": "application/json" },

          body: JSON.stringify({ claim })

        });


        const data = await response.json();


        if (!response.ok) {

          resultsDiv.innerHTML = `<div class="errorBox">

            <strong>Request failed.</strong>

            <div class="muted">${data.detail || "Unknown error"}</div>

          </div>`;

          return;

        }


        if (!data.reviews || data.reviews.length === 0) {

          resultsDiv.innerHTML = `<p class="no-result">No fact-checks found.</p>`;

          return;

        }


        resultsDiv.innerHTML = "";


        data.reviews.forEach((review) => {

          const isAI = (review.publisher || "").toLowerCase().includes("truescope ai");

          const safeUrl = review.url && review.url !== "#" ? review.url : "";

          const pillClass = ratingClass(review.rating);


          resultsDiv.innerHTML += `

            <div class="card ${isAI ? "ai-card" : ""}">

              <div class="cardTop">

                <h3 class="cardTitle">

                  ${review.publisher}

                  ${isAI ? `<span class="badge ai">AI Summary</span>` : ""}

                </h3>

                <span class="rating-pill ${pillClass}">${safeShortRating(review.rating)}</span>

              </div>


              <p class="cardText">${(review.text || "").replaceAll("<", "&lt;").replaceAll(">", "&gt;")}</p>


              ${safeUrl ? `<a class="cardLink" href="${safeUrl}" target="_blank" rel="noopener">Read more</a>` : ""}

            </div>

          `;

        });


      } catch (err) {

        console.error(err);

        resultsDiv.innerHTML = `<p class="no-result">Something went wrong. Check backend is running.</p>`;

      }

    }


    document.getElementById("checkBtn").addEventListener("click", checkClaim);

    document.getElementById("claimInput").addEventListener("keydown", (e) => {

      if (e.key === "Enter") checkClaim();

    });

  </script>
